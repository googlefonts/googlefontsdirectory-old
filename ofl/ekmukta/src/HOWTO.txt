## subset the latin and latin-ext - Regular
pyftsubset EkMukta-Regular.ttf \
$(<~/src/googlefontdirectory/tools/encodings/unicodes-latin.txt) \
--layout-features='' --glyph-names --symbol-cmap \
--notdef-glyph --notdef-outline --recommended-glyphs \
--name-IDs='*' --name-languages='*' \
--hinting;
mv EkMukta-Regular.ttf.subset EkMukta-Regular.latin;

pyftsubset EkMukta-Regular.ttf \
$(<~/src/googlefontdirectory/tools/encodings/unicodes-latin.txt) \
$(<~/src/googlefontdirectory/tools/encodings/unicodes-latin-ext.txt) \
--layout-features='' --glyph-names --symbol-cmap \
--notdef-glyph --notdef-outline --recommended-glyphs \
--name-IDs='*' --name-languages='*' \
--verbose --timing --hinting;
mv EkMukta-Regular.ttf.subset EkMukta-Regular.latin-ext;

pyftsubset EkMukta-Regular.ttf \
--text=\ Ek\ Mukta \
--layout-features='' --glyph-names --symbol-cmap \
--notdef-glyph --notdef-outline --recommended-glyphs \
--name-IDs='*' --name-languages='*' \
--verbose --timing --hinting;
mv EkMukta-Regular.ttf.subset EkMukta-Regular.menu;
