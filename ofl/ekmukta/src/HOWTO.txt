cd ~/src/googlefontdirectory/ofl/ekmukta/;

family="EkMukta"
menutext="Ek Mukta"

for style in Regular Bold ExtraBold ExtraLight Light Medium SemiBold; do 

    pyftsubset $family-$style.ttf \
    $(<~/src/googlefontdirectory/tools/encodings/unicodes-latin.txt) \
    --notdef-outline --recommended-glyphs \
    --name-IDs='*' \
    --hinting;
    mv $family-$style.ttf.subset $family-$style.latin;

    pyftsubset $family-$style.ttf \
    $(<~/src/googlefontdirectory/tools/encodings/unicodes-latin.txt) \
    $(<~/src/googlefontdirectory/tools/encodings/unicodes-latin-ext.txt) \
    --notdef-outline --recommended-glyphs \
    --name-IDs='*' \
    --verbose --timing --hinting;
    mv $family-$style.ttf.subset $family-$style.latin-ext;

    pyftsubset $family-$style.ttf \
    --text="$menutext" \
    --notdef-outline --recommended-glyphs \
    --name-IDs='*' \
    --verbose --timing --hinting;
    mv $family-$style.ttf.subset $family-$style.menu;

    pyftsubset $family-$style.ttf \
    $(<~/src/googlefontdirectory/tools/encodings/unicodes-latin.txt) \
    $(<~/src/googlefontdirectory/tools/encodings/unicodes-devanagari.txt) \
    --notdef-outline --recommended-glyphs \
    --name-IDs='*' \
    --verbose --timing --hinting;
    mv $family-$style.ttf.subset $family-$style.devanagari;

done
