cd ~/src/googlefontdirectory/ofl/encodesansnarrow/;

# fix filenames

mv EncodeSansNarrow-100-Thin.ttf        EncodeSansNarrow-Thin.ttf
mv EncodeSansNarrow-200-ExtraLight.ttf  EncodeSansNarrow-ExtraLight.ttf
mv EncodeSansNarrow-300-Light.ttf       EncodeSansNarrow-Light.ttf
mv EncodeSansNarrow-400-Regular.ttf     EncodeSansNarrow-Regular.ttf
mv EncodeSansNarrow-500-Medium.ttf      EncodeSansNarrow-Medium.ttf
mv EncodeSansNarrow-600-SemiBold.ttf    EncodeSansNarrow-SemiBold.ttf
mv EncodeSansNarrow-700-Bold.ttf        EncodeSansNarrow-Bold.ttf
mv EncodeSansNarrow-800-ExtraBold.ttf   EncodeSansNarrow-ExtraBold.ttf
mv EncodeSansNarrow-900-Black.ttf       EncodeSansNarrow-Black.ttf

# fix nbsp

for font in `ls -1 *ttf`; do ~/src/fontbakery/scripts/fix-ttf-nbsp.py $font; done
rm *ttf;
rename s/\.fix//g *;

family="EncodeSansNarrow"
menutext="Encode Sans Narrow"

for style in Thin ExtraLight Light Regular Medium SemiBold Bold ExtraBold Black; do 

    pyftsubset $family-$style.ttf \
    $(<~/src/googlefontdirectory/tools/encodings/unicodes-latin.txt) \
    --notdef-outline --recommended-glyphs \
    --name-IDs='*' \
    --hinting;
    mv $family-$style.ttf.subset $family-$style.latin;

    pyftsubset $family-$style.ttf \
    $(<~/src/googlefontdirectory/tools/encodings/unicodes-latin.txt) \
    $(<~/src/googlefontdirectory/tools/encodings/unicodes-latin-ext.txt) \
    --notdef-outline --recommended-glyphs \
    --name-IDs='*' \
    --verbose --timing --hinting;
    mv $family-$style.ttf.subset $family-$style.latin-ext;

    pyftsubset $family-$style.ttf \
    --text="$menutext" \
    --notdef-outline --recommended-glyphs \
    --name-IDs='*' \
    --verbose --timing --hinting;
    mv $family-$style.ttf.subset $family-$style.menu;

done;

cp ../encodesanswide/DESC* .;

~/src/fontbakery/scripts/genmetadata.py .;
~/src/fontbakery/scripts/genmetadata.py .;
mv METADATA.json.new METADATA.json;
mate ../encodesanswide/METADATA.json METADATA.json DES*;
