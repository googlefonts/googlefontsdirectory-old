cd ~/src/googlefontdirectory/ofl/lakkireddy/;

family="LakkiReddy"
menuchars=`~/googlefontdirectory/tools/chars/string-line.py $family`

for style in Regular; do

    pyftsubset $family-$style.ttf \
	--unicodes=" \
    $( cat ~/src/googlefontdirectory/tools/encodings/latin_unique-glyphs.nam | cut -d\  -f1 )" \
    --notdef-outline \
    --name-IDs='*' \
    --verbose --timing --hinting;
    mv $family-$style.ttf.subset $family-$style.latin;

    pyftsubset $family-$style.ttf \
	--unicodes=" \
    $( cat ~/src/googlefontdirectory/tools/encodings/latin_unique-glyphs.nam | cut -d\  -f1 ) \
    $( cat ~/src/googlefontdirectory/tools/encodings/telugu_unique-glyphs.nam | cut -d\  -f1 )" \
    --notdef-outline \
    --name-IDs='*' \
    --verbose --timing --hinting;
    mv $family-$style.ttf.subset $family-$style.telugu;
   
    pyftsubset $family-$style.ttf \
    --unicodes="U+0020 $menuchars \
    $(<~/src/googlefontdirectory/tools/encodings/unicodes-telugu-menu.txt)" \
    --notdef-outline \
    --name-IDs='*' \
    --verbose --timing --hinting;
    mv $family-$style.ttf.subset $family-$style.menu;

done
