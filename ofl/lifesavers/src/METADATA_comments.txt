cd ~/googlefontdirectory/ofl/; \
  for family in lifesavers; do \
    for font in `ls -1 $family/*ttf | cut -d. -f1`; do \
      cd ~/src/font-optimizer; \
      ./subset.pl -v --licensesubst="SIL OFL 1.1" \
      --exclude="c2sc,smcp,ordn,sups,sinf,subs,dlig,frac,ss01,aalt,kern,liga,salt" \
       --chars="Life Savers" \
      ~/googlefontdirectory/ofl/$font.{ttf,menu}; \
      cd ~/googlefontdirectory/ofl/; \
      ~/googlefontdirectory/tools/chars/showchars.py \
       $font.menu; 
    cd ~/googlefontdirectory/ofl/lifesavers/; \
  done;
done;

   for font in `ls -1 $family/*ttf | cut -d. -f1`; do \
       for subset in \
           latin latin-ext; do \
           echo subset.py --null --nmr --roundtrip --script \

               --null --nmr --roundtrip --script --subset=$subset \
               $font.ttf $font.$subset >> $font.$subset.log \
               2>> $font.$subset.log; \
       done; \
   done; \
done; \
